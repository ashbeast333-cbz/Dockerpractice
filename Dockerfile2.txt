FROM ubuntu:20.04
ENV DEBIAN_FRONTEND=noninteractive

RUN apt-get update && \
    apt-get install -y apache2 unzip curl && \
    rm -rf /var/lib/apt/lists/*

# Download and extract template
RUN curl -L https://templatemo.com/tm-578-first-portfolio.zip -o /tmp/template.zip && \
    unzip /tmp/template.zip -d /tmp/template && \
    cp -r /tmp/template/* /var/www/html/ && \
    rm -rf /tmp/template /tmp/template.zip

EXPOSE 80
CMD ["/usr/sbin/apache2ctl", "-D", "FOREGROUND"]
